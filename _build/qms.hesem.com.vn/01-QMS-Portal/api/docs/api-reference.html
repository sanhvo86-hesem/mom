1. Check Auth Status

# Check if the current session is authenticated
curl -s -b cookies.txt \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/?action=status"

2. Login

# Login with username and password (saves session cookie)
curl -s -c cookies.txt \
  -X POST \
  -H "Content-Type: application/json" \
  -d '{"username":"john.doe","password":"MySecureP@ss1"}' \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/?action=auth_login"

3. MFA Verify

# Submit TOTP code after login (when mfa_required: true)
curl -s -b cookies.txt -c cookies.txt \
  -X POST \
  -H "Content-Type: application/json" \
  -d '{"code":"123456"}' \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/?action=auth_mfa_verify"

4. List Documents

# Get all documents visible to the current user
curl -s -b cookies.txt \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/documents"

5. Create a Document

# Create a new document (requires CSRF token from status/login response)
curl -s -b cookies.txt \
  -X POST \
  -H "Content-Type: application/json" \
  -H "X-CSRF-Token: YOUR_CSRF_TOKEN" \
  -d '{
    "code": "SOP-999",
    "title": "New Procedure Title",
    "cat": "SOP",
    "owner": "QA/QMS"
  }' \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/documents"

6. List Online Forms

# List all online-enabled forms
curl -s -b cookies.txt \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/forms"

7. Submit a Form Entry

# Submit an entry for form FRM-631
curl -s -b cookies.txt \
  -X POST \
  -H "Content-Type: application/json" \
  -H "X-CSRF-Token: YOUR_CSRF_TOKEN" \
  -d '{
    "code": "FRM-631",
    "data": {
      "ncr_number": "NCR-2025-042",
      "description": "Surface finish out of spec",
      "severity": "major"
    }
  }' \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/forms/FRM-631/entries"

8. Generate Record ID

# Generate the next NCR record ID (e.g. NCR-2025-043)
curl -s -b cookies.txt \
  -X POST \
  -H "Content-Type: application/json" \
  -H "X-CSRF-Token: YOUR_CSRF_TOKEN" \
  -d '{"type":"NCR","digits":3}' \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/records/next-id"

9. Scan Folders

# Scan document tree (returns docs + folder structure)
curl -s -b cookies.txt \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/folders"

10. List Users (Admin)

# List all user accounts (admin only)
curl -s -b cookies.txt \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/users"

11. Git Sync (Admin)

# Commit and push all document changes to git
curl -s -b cookies.txt \
  -X POST \
  -H "X-CSRF-Token: YOUR_CSRF_TOKEN" \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/admin/git/sync"

12. Dictionary - List Terms

# Get all QMS glossary terms
curl -s -b cookies.txt \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/dictionary"

13. Logout

# Destroy session
curl -s -b cookies.txt \
  -X POST \
  -H "X-CSRF-Token: YOUR_CSRF_TOKEN" \
  "https://qms.hesem.com.vn/01-QMS-Portal/api/?action=auth_logout"
